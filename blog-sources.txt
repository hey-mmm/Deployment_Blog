---
##创建namespace
apiVersion: v1
kind: Namespace
metadata:
  name: blog
---
##创建blog的pv
apiVersion: v1
kind: PersistentVolume
metadata:
  name: typecho-pv
spec:
  capacity:
    storage: 1Gi
  accessModes:
    - ReadWriteMany
  persistentVolumeReclaimPolicy: Retain
  storageClassName: blog
  hostPath:
    path: /root/k8s/blog/data/typecho
---
##创建mysql的pv
apiVersion: v1
kind: PersistentVolume
metadata:
  name: mysql-pv
spec:
  capacity:
    storage: 1Gi
  accessModes:
    - ReadWriteMany
  persistentVolumeReclaimPolicy: Retain
  storageClassName: blog
  hostPath:
    path: /root/k8s/blog/data/mysql

---

apiVersion: v1
kind: ConfigMap
metadata:
  name: typecho-config
  namespace: blog
data:
  TYPECHO_DATABASE: "mysql:host=blog-mysql;port=3306;dbname=typecho"
  TYPECHO_TIMEZONE: "Asia/Shanghai"

---

apiVersion: v1
kind: Secret
metadata:
  name: mysql-secret
  namespace: blog
type: Opaque

  MYSQL_ROOT_PASSWORD: MQ==

  MYSQL_USER: cm9vdA==
